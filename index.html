  <!--

    =========================================================

    Scripts (JavaScript)

    =========================================================

    - Se utilizan funciones separadas para:

      * Generar contraseñas

      * Actualizar UI

      * Manejar localStorage

      * Controlar el tema (oscuro/claro)

    - Se usa crypto.getRandomValues para mayor seguridad en la generación.

    - Se incluyen comentarios explicativos en secciones clave.

  -->

  <script>

    // DOM elements

    const passwordLengthInput = document.getElementById('password-length');

    const includeUppercaseInput = document.getElementById('include-uppercase');

    const includeLowercaseInput = document.getElementById('include-lowercase');

    const includeNumbersInput = document.getElementById('include-numbers');

    const includeSymbolsInput = document.getElementById('include-symbols');

    const generateBtn = document.getElementById('generate-btn');

    const copyBtn = document.getElementById('copy-btn');

    const copyFeedback = document.getElementById('copy-feedback');

    const passwordDisplay = document.getElementById('password-display');

    const passwordStrengthIndicator = document.getElementById('password-strength-indicator');

    const saveBtn = document.getElementById('save-btn');

    const savedPasswordsList = document.getElementById('saved-passwords-list');

    const toggleThemeBtn = document.getElementById('toggle-theme-btn');

    const bodyElement = document.body;


    // Caracteres disponibles

    const UPPERCASE_CHARS = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';

    const LOWERCASE_CHARS = 'abcdefghijklmnopqrstuvwxyz';

    const NUMBERS_CHARS = '0123456789';

    const SYMBOLS_CHARS = '!@#$%^&*()_+~`|}{[]\\:;?><,./-=';


    // Al cargar la página, inicializamos tema y mostramos las contraseñas almacenadas

    window.addEventListener('DOMContentLoaded', () => {

      loadThemePreference();

      displaySavedPasswords();

    });


    // Función para generar contraseñas usando crypto.getRandomValues

    const generatePassword = (length, useUppercase, useLowercase, useNumbers, useSymbols) => {

      let chars = '';

      if (useUppercase) chars += UPPERCASE_CHARS;

      if (useLowercase) chars += LOWERCASE_CHARS;

      if (useNumbers) chars += NUMBERS_CHARS;

      if (useSymbols) chars += SYMBOLS_CHARS;


      // Manejo de error: si no se selecciona ningún tipo de carácter

      if (!chars) {

        alert('Por favor, selecciona al menos una opción (mayúsculas, minúsculas, números o símbolos).');

        return '';

      }


      // Generar arreglo de bytes aleatorios

      const array = new Uint8Array(length);

      crypto.getRandomValues(array);


      // Construir la contraseña tomando cada byte mod la longitud de chars

      let password = '';

      for (let i = 0; i < length; i++) {

        const randomIndex = array[i] % chars.length;

        password += chars.charAt(randomIndex);

      }


      return password;

    };


    // Función para actualizar el indicador de fortaleza de la contraseña

    const updateStrengthIndicator = (password) => {

      if (!password) {

        passwordStrengthIndicator.textContent = 'Fortaleza';

        passwordStrengthIndicator.classList.remove('strength-weak', 'strength-medium', 'strength-strong');

        return;

      }


      let score = 0;


      // Aumentamos el puntaje según la longitud

      score += password.length * 4;


      // Aumentamos el puntaje según variedad de caracteres

      const hasUpper = /[A-Z]/.test(password);

      const hasLower = /[a-z]/.test(password);

      const hasNumbers = /[0-9]/.test(password);

      const hasSymbols = /[^A-Za-z0-9]/.test(password);


      if (hasUpper) score += 5;

      if (hasLower) score += 5;

      if (hasNumbers) score += 5;

      if (hasSymbols) score += 5;


      // Evaluamos el score para clasificar fortaleza

      let strength = 'Débil';

      passwordStrengthIndicator.classList.remove('strength-weak', 'strength-medium', 'strength-strong');


      if (score < 40) {

        strength = 'Débil';

        passwordStrengthIndicator.classList.add('strength-weak');

      } else if (score >= 40 && score < 70) {

        strength = 'Media';

        passwordStrengthIndicator.classList.add('strength-medium');

      } else {

        strength = 'Fuerte';

        passwordStrengthIndicator.classList.add('strength-strong');

      }


      passwordStrengthIndicator.textContent = `Fortaleza: ${strength}`;

    };


    // Evento: hacer clic en "Generar"

    generateBtn.addEventListener('click', () => {

      const length = parseInt(passwordLengthInput.value, 10);

      const useUppercase = includeUppercaseInput.checked;

      const useLowercase = includeLowercaseInput.checked;

      const useNumbers = includeNumbersInput.checked;

      const useSymbols = includeSymbolsInput.checked;


      const newPassword = generatePassword(length, useUppercase, useLowercase, useNumbers, useSymbols);

      passwordDisplay.textContent = newPassword;

      updateStrengthIndicator(newPassword);

      hideCopyFeedback();

    });


    // Evento: hacer clic en "Copiar"

    copyBtn.addEventListener('click', () => {

      const passwordToCopy = passwordDisplay.textContent.trim();

      if (!passwordToCopy) return;


      // Copiar al portapapeles

      navigator.clipboard.writeText(passwordToCopy).then(() => {

        showCopyFeedback();

      });

    });


    // Función para mostrar feedback "¡Copiado!"

    const showCopyFeedback = () => {

      copyFeedback.classList.add('visible');

      setTimeout(() => {

        hideCopyFeedback();

      }, 2000); // Ocultar después de 2 segundos

    };


    // Función para ocultar feedback

    const hideCopyFeedback = () => {

      copyFeedback.classList.remove('visible');

    };


    // Evento: hacer clic en "Guardar"

    saveBtn.addEventListener('click', () => {

      const currentPassword = passwordDisplay.textContent.trim();

      if (!currentPassword) {

        alert('Primero genera una contraseña antes de guardar.');

        return;

      }


      savePassword(currentPassword);

      displaySavedPasswords();

    });


    // Función para guardar contraseñas en localStorage

    const savePassword = (password) => {

      const storedPasswords = JSON.parse(localStorage.getItem('savedPasswords')) || [];

      storedPasswords.push(password);

      localStorage.setItem('savedPasswords', JSON.stringify(storedPasswords));

    };


    // Función para mostrar contraseñas guardadas

    const displaySavedPasswords = () => {

      const storedPasswords = JSON.parse(localStorage.getItem('savedPasswords')) || [];

      savedPasswordsList.innerHTML = '';


      storedPasswords.forEach((pass, index) => {

        const li = document.createElement('li');

        li.innerHTML = `

          <span>${pass}</span>

          <div>

            <button class="btn" data-copy-index="${index}">Copiar</button>

            <button class="btn btn-delete" data-delete-index="${index}">Eliminar</button>

          </div>

        `;

        savedPasswordsList.appendChild(li);

      });


      // Agregar eventos a botones de cada ítem

      document.querySelectorAll('[data-copy-index]').forEach((btn) => {

        btn.addEventListener('click', (e) => {

          const idx = e.target.getAttribute('data-copy-index');

          copySavedPassword(idx);

        });

      });


      document.querySelectorAll('[data-delete-index]').forEach((btn) => {

        btn.addEventListener('click', (e) => {

          const idx = e.target.getAttribute('data-delete-index');

          deleteSavedPassword(idx);

        });

      });

    };


    // Función para copiar una contraseña guardada

    const copySavedPassword = (index) => {

      const storedPasswords = JSON.parse(localStorage.getItem('savedPasswords')) || [];

      if (storedPasswords[index]) {

        navigator.clipboard.writeText(storedPasswords[index]).then(() => {

          alert('¡Contraseña copiada de la lista!');

        });

      }

    };


    // Función para eliminar una contraseña guardada

    const deleteSavedPassword = (index) => {

      let storedPasswords = JSON.parse(localStorage.getItem('savedPasswords')) || [];

      storedPasswords.splice(index, 1);

      localStorage.setItem('savedPasswords', JSON.stringify(storedPasswords));

      displaySavedPasswords();

    };


    // Tema oscuro / claro

    toggleThemeBtn.addEventListener('click', () => {

      const currentTheme = bodyElement.classList.contains('light-theme') ? 'light' : 'dark';

      const newTheme = currentTheme === 'light' ? 'dark' : 'light';

      bodyElement.classList.toggle('light-theme');

      bodyElement.classList.toggle('dark-theme');

      // Guardar preferencia

      localStorage.setItem('preferredTheme', newTheme);

    });


    // Cargar la preferencia de tema

    const loadThemePreference = () => {

      const preferredTheme = localStorage.getItem('preferredTheme');

      if (preferredTheme === 'dark') {

        bodyElement.classList.remove('light-theme');

        bodyElement.classList.add('dark-theme');

      } else {

        // Por defecto, tema claro

        bodyElement.classList.remove('dark-theme');

        bodyElement.classList.add('light-theme');

      }

    };

  </script>
